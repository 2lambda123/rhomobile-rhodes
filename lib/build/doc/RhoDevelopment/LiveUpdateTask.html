def execute
  puts "Executing #{self.class.name} at #{Time::now}".primary
  begin
    self.action
  rescue => e
    puts "Executing #{self.class.taskName} failed".warning
    puts e.inspect.to_s.info
    puts e.backtrace.to_s.info
  end
end
